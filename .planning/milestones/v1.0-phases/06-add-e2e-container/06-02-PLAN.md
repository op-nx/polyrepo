Rewrite the test structure:

```typescript
describe('@op-nx/polyrepo', () => {
  let container: StartedTestContainer;

  beforeAll(async () => {
    const snapshotImage = inject('snapshotImage');
    container = await new GenericContainer(snapshotImage)
      .withCommand(['sleep', 'infinity'])
      .start();
  });

  afterAll(async () => {
    if (container) {
      await container.stop();
    }
  });

  it('should be installed', async () => {
    const { exitCode } = await container.exec(
      ['npm', 'ls', '@op-nx/polyrepo'],
      { workingDir: '/workspace' },
    );
    expect(exitCode).toBe(0);
  });

  describe('polyrepo-status', () => {
    beforeAll(async () => {
      // Register plugin in nx.json using container.exec + sh -c
      // Write nx.json with plugin config referencing /repos/nx as local path repo
      // Use container.exec(['sh', '-c', 'cat > /workspace/nx.json << ...'])
      // or use container.copyContentToContainer() if available
      //
      // The nx.json content should be:
      // {
      //   "plugins": [{
      //     "plugin": "@op-nx/polyrepo",
      //     "options": {
      //       "repos": {
      //         "nx": { "url": "/repos/nx", "depth": 1, "ref": "master" }
      //       }
      //     }
      //   }]
      // }
      //
      // IMPORTANT: Use local path "/repos/nx" (prebaked in Docker image),
      // NOT the GitHub URL. This eliminates network dependency at test time.
    });

    it('should report unsynced repos', async () => {
      const { stdout } = await container.exec(
        ['npx', 'nx', 'polyrepo-status'],
        { workingDir: '/workspace' },
      );
      expect(stdout).toContain('[not synced]');
      expect(stdout).toContain('1 configured, 0 synced, 1 not synced');
    });

    it('should register target on root project', async () => {
      const { stdout } = await container.exec(
        ['npx', 'nx', 'show', 'project', '@org/source', '--json'],
        { workingDir: '/workspace' },
      );
      // Strip any Nx warnings before JSON (same pattern as current test)
      const jsonStr = stdout.replace(/^[^{]*/, '');
      const project = JSON.parse(jsonStr);
      expect(project.targets['polyrepo-status']).toBeDefined();
      expect(project.targets['polyrepo-status'].executor).toBe(
        '@op-nx/polyrepo:status',
      );
    });
  });
});
```
