export const createDependencies: CreateDependencies<PolyrepoConfig> = async (
  options,
  context,
) => {
  const dependencies: RawProjectGraphDependency[] = [];
  let report: PolyrepoGraphReport | undefined;

  try {
    const config = validateConfig(options);
    const optionsHash = hashObject(options ?? {});
    report = await populateGraphReport(
      config,
      context.workspaceRoot,
      optionsHash,
    );
  } catch {
    return dependencies;
  }

  for (const [, repoReport] of Object.entries(report.repos)) {
    for (const dep of repoReport.dependencies) {
      if (context.projects[dep.source] && context.projects[dep.target]) {
        dependencies.push({
          source: dep.source,
          target: dep.target,
          type: DependencyType.implicit,
        });
      }
    }
  }

  return dependencies;
};

**index.ts changes:**

1. Add import: `import { detectCrossRepoDependencies } from './lib/graph/detect';`
2. Restructure `createDependencies` to separate extraction try/catch from detection. The key change: `config` and `report` must be available after the try/catch so detection runs OUTSIDE it. Per user decision, OVRD-03 errors must propagate.

   New structure:

   ```typescript
   export const createDependencies: CreateDependencies<PolyrepoConfig> = async (
     options,
     context,
   ) => {
     const dependencies: RawProjectGraphDependency[] = [];

     let report: PolyrepoGraphReport | undefined;
     let config: PolyrepoConfig;

     try {
       config = validateConfig(options);
       const optionsHash = hashObject(options ?? {});

       report = await populateGraphReport(
         config,
         context.workspaceRoot,
         optionsHash,
       );
     } catch {
       // If extraction fails, return no dependencies (degraded mode)
       return dependencies;
     }

     // Intra-repo edges (existing behavior)
     for (const [, repoReport] of Object.entries(report.repos)) {
       for (const dep of repoReport.dependencies) {
         if (context.projects[dep.source] && context.projects[dep.target]) {
           dependencies.push({
             source: dep.source,
             target: dep.target,
             type: DependencyType.implicit,
           });
         }
       }
     }

     // Cross-repo edges -- NOT wrapped in try/catch
     // OVRD-03 validation errors intentionally propagate to Nx
     const crossRepoDeps = detectCrossRepoDependencies(report, config, context);
     dependencies.push(...crossRepoDeps);

     return dependencies;
   };
   ```

   Note: `config` is declared with `let config: PolyrepoConfig;` outside the try block. TypeScript's definite assignment analysis will be satisfied because the catch block returns early, guaranteeing `config` is assigned if execution reaches the detection call.

**Important:** The `validateConfig` call is INSIDE the try/catch. If config validation fails (e.g., malformed options), the function degrades gracefully and returns `[]`. This matches the existing behavior. Only detection errors (OVRD-03) propagate.
